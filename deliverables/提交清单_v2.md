```bash
export QWEN_MODEL=qwen3.7-plus
export QWEN_ENABLE_THINKING=false
PYTHONPATH=backend .venv/bin/python -m uvicorn miniso_studio.starter.api:app \
  --host 127.0.0.1 --port 8767 > /tmp/trend2sku-online-smoke.log 2>&1 &
SMOKE_PID=$!
curl -s http://127.0.0.1:8767/api/health > /tmp/trend2sku-online-health.json
curl -s -X POST http://127.0.0.1:8767/api/run \
  -H 'Content-Type: application/json' \
  -d '{"brief":"在线叙述脱敏验证","product_category":"stationery","custom_category":"","target_segment":"student","target_market":"china","price_band":"entry","ip_strategy":"original","objectives":["social","margin"],"constraints":"不改变数值规则","hitl":false,"thread_id":"online-smoke-001"}' \
  > /tmp/trend2sku-online-result.json
! rg -q -F -- "$QWEN_API_KEY" \
  /tmp/trend2sku-online-smoke.log \
  /tmp/trend2sku-online-health.json \
  /tmp/trend2sku-online-result.json
kill "$SMOKE_PID"
wait "$SMOKE_PID" || true
```
